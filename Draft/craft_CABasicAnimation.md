# CABasicAnimation

[CABasicAnimation animationWithKeyPath:@"transform.translation.y"];

animationWithKeyPath的值：

    transform.scale = 比例轉換

    transform.scale.x = 闊的比例轉換

    transform.scale.y = 高的比例轉換

    transform.rotation.z = 平面圖的旋轉

    opacity = 透明度

    margin

    zPosition

    backgroundColor    背景颜色

    cornerRadius    圆角
